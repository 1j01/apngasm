# shared library and sources
ADD_LIBRARY(apngasm SHARED
#ADD_LIBRARY(apngasm-static STATIC
  ${APNGASM_SOURCE_DIR}/src/apngasm.cpp
  ${APNGASM_SOURCE_DIR}/src/apngframe.cpp
  ${APNGASM_SOURCE_DIR}/src/spec/specreader.cpp
  ${APNGASM_SOURCE_DIR}/src/spec/specwriter.cpp
  ${APNGASM_SOURCE_DIR}/src/spec/priv/specreaderimpl.cpp
  ${APNGASM_SOURCE_DIR}/src/spec/priv/specwriterimpl.cpp
  ${APNGASM_SOURCE_DIR}/src/listener/apngasmlistener.cpp
)

FIND_PACKAGE(PNG REQUIRED)
INCLUDE_DIRECTORIES(${PNG_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(apngasm ${PNG_LIBRARY})

#find_package(ZLIB_REQUIRED)
INCLUDE_DIRECTORIES(${ZLIB_INCLUDE_DIRS})
TARGET_LINK_LIBRARIES(apngasm ${ZLIB_LIBRARIES})

FIND_PACKAGE(Boost REQUIRED COMPONENTS regex filesystem system) 
INCLUDE_DIRECTORIES(${Boost_INCLUDE_DIRS})
TARGET_LINK_LIBRARIES(apngasm ${Boost_LIBRARIES})

SET(CPACK_OUTPUT_CONFIG_FILE "${CMAKE_CURRENT_SOURCE_DIR}/CPackConfig.cmake")
SET(CPACK_DEB_COMPONENT_INSTALL ON)
INSTALL(TARGETS apngasm
	LIBRARY DESTINATION lib
	COMPONENT libraries)

INSTALL(FILES ../src/apngasm.h ../src/apngframe.h ../src/apngasm-version.h
	DESTINATION include
	COMPONENT headers)


##CPack definitions for package creation
INCLUDE(InstallRequiredSystemLibraries)

SET(CPACK_PACKAGE_VERSION_MAJOR "${VERSION_MAJOR}")
SET(CPACK_PACKAGE_VERSION_MINOR "${VERSION_MINOR}")
SET(CPACK_PACKAGE_VERSION_PATCH "${VERSION_PATCH}")
SET(CPACK_PACKAGE_DESCRIPTION_SUMMARY "apngasm APNG assembler libraries")
SET(CPACK_PACKAGE_DESCRIPTION_FILE "${CMAKE_CURRENT_SOURCE_DIR}/../DESCRIPTION.txt")
SET(CPACK_PACKAGE_LICENSE_FILE "${CMAKE_CURRENT_SOURCE_DIR}/../LICENSE.txt")
SET(CPACK_PACKAGE_CONTACT "zero@genshin.org")
SET(CPACK_PACKAGE_FILE_NAME "libapngasm_${VERSION}")
SET(CPACK_COMPONENTS_ALL libraries headers)

IF(UNIX)
	SET(CPACK_GENERATOR "DEB")
	SET(CPACK_DEBIAN_PACKAGE_DEPENDS "libc6 (>= 2.14), libpng12-0 (>= 1.2.13-4), zlib1g (>= 1:1.1.4)")
ENDIF(UNIX)
IF(APPLE)
	SET(CPACK_GENERATOR "PackageMaker")
ENDIF(APPLE)
IF(WIN32)
	SET(CPACK_GENERATOR "NSIS")
ENDIF(WIN32)

INCLUDE(CPack)
